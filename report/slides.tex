% ── SLIDE 5 · Corpus ─────────────────────────────────────────
\begin{frame}{A Trilingual Corpus — From Scratch}

  \begin{columns}[T]
    \column{0.40\textwidth}
      \begin{block}{Corpus at a glance}
        \centering
        \vspace{4pt}
        \begin{tabular}{lrr}
          \toprule
          Lang & Texts & Hate\% \\
          \midrule
          \textbf{EN} & 58,940 & mixed \\
          \textbf{ZH} & 1,965  & 100\% \\
          \textbf{JA} & 2,360  & 100\% \\
          \midrule
          \textbf{Total} & \textbf{63,265} & \\
          \bottomrule
        \end{tabular}
        \vspace{4pt}
      \end{block}
      \vspace{6pt}
      \footnotesize
      \textbf{EN:} filtered from 4 public datasets\\
      (HateXplain, Jigsaw, MLMA, MHS)\\[4pt]
      \textbf{ZH/JA:} crawled + manually annotated\\
      (Tieba / 5ch)
    \column{0.56\textwidth}
      \begin{alertblock}{The low-resource challenge}
        ZH and JA data are scarce. Classifiers trained naively perform \textbf{very poorly}.
      \end{alertblock}
      \vspace{6pt}
      \begin{block}{Two-stage pipeline for ZH / JA}
        \begin{enumerate}
          \item \textbf{Native crawl} → manual annotation → fine-tuned classifier
          \item \textbf{LLM-based localization}: translate EN seed texts into ZH/JA → filter with same classifier
        \end{enumerate}
      \end{block}
      \vspace{4pt}
      \footnotesize
      \textit{Key finding:} LLM seed-guided generation collapses into\\
      \textbf{keyword loops} after 20–30 iterations\\
      (ZH: 90\% redundancy at 50 samples; JA: 100\% at 30)\\
      $\Rightarrow$ back-translation is more robust.
  \end{columns}
\end{frame}

            (Strategy C,0.635)
          };
          % Japanese
          \addplot[fill=accent!80,draw=accent] coordinates {
            (Baseline,0.294)
            (Strategy B,0.42)
            (Strategy C,0.579)
          };
          \legend{Chinese (ZH), Japanese (JA)}
        \end{axis}
      \end{tikzpicture}
    \column{0.44\textwidth}
      \vspace{4pt}
      \begin{block}{Strategy A — Baseline}
        Train on native data only. \\
        \textit{Very low F1 due to data scarcity.}
      \end{block}
      \vspace{4pt}
      \begin{block}{Strategy B — Train-only BT}
        Back-translate training set only.\\
        \textit{Distribution shift between train \& eval degrades performance.}
      \end{block}
      \vspace{4pt}
      \begin{alertblock}{\faCheckCircle\ Strategy C — Double BT (Winner)}
        Back-translate \textbf{both} train \& eval sets.\\
        Eliminates distribution gap.\\
        \highlight{ZH: $0.172 \to 0.635$} \quad
        \highlight{JA: $0.294 \to 0.579$}
      \end{alertblock}
  \end{columns}
\end{frame}

\section{Discussion}

% ── SLIDE 15 · Discussion – two regularities ─────────────────
\begin{frame}{Discussion: What Do These Patterns Tell Us?}

  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{block}{\#1 — Attack Frames as Genre Conventions}
        \small
        Frame choice is \textbf{not random} and \textbf{not language-specific}.\\[4pt]
        Each target type attracts a predictable set of frames — like a \emph{rhetorical genre}.\\[4pt]
        Implication: hate speech has \textbf{stable structural grammar} across cultures.
      \end{block}
    \column{0.48\textwidth}
      \begin{block}{\#2 — Two Moral Architectures}
        \small
        EN discourse: \emph{moral transgressor} model — Christianity is judged by universal ethics.\\[4pt]
        ZH/JA discourse: \emph{outgroup threat} model — Christianity is a cultural danger.\\[4pt]
        Implication: binary classifiers collapse this difference; structured analysis reveals it.
      \end{block}
  \end{columns}

  \vspace{10pt}
  \begin{center}
    \includegraphics[width=0.55\textwidth]{disc_genre_convention_map.png}
  \end{center}
\end{frame}

% ── SLIDE 16 · Methodological reflections ────────────────────
\begin{frame}{Discussion: Methodological Reflections}

  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{block}{What worked well}
        {\small
        \begin{itemize}
          \item \textbf{Double back-translation} resolves distribution shift — a reusable solution for low-resource hate speech tasks
          \item \textbf{Cross-lingual centroid alignment} in BERTopic produces genuinely mixed topics without translation
          \item \textbf{Corpus-induction} (not taxonomy imposition) lets Japanese Unification Church discourse emerge naturally
        \end{itemize}}
      \end{block}
    \column{0.48\textwidth}
      \begin{alertblock}{What to watch for}
        {\small
        \begin{itemize}
          \item \textbf{LLM data collapse}: after 20–30 iterations, generated ZH/JA text becomes 90–100\% redundant — a real problem for low-resource augmentation
          \item \textbf{Platform bias}: Tieba (ZH) and 5ch (JA) are not representative of all Chinese/Japanese internet speech
          \item \textbf{Annotation subjectivity}: implicit hate is inherently harder to label consistently
        \end{itemize}}
      \end{alertblock}
  \end{columns}
\end{frame}

% ── SLIDE 18 · Limitations & Future Work ─────────────────────
\begin{frame}{Limitations \& Future Work}

  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{alertblock}{Current limitations}
        \begin{itemize}
          \item \textbf{Platform coverage}: Tieba \& 5ch may not generalise to WeChat, Twitter/X, LINE
          \item \textbf{Temporal snapshot}: discourse evolves; pipeline should be re-run on fresh data
          \item \textbf{ZH/JA corpus size}: still relatively small despite augmentation
          \item \textbf{Annotation}: implicit hate labels are noisy; inter-annotator agreement is moderate
        \end{itemize}
      \end{alertblock}
    \column{0.48\textwidth}
      \begin{block}{Promising directions}
        \begin{itemize}
          \item Extend pipeline to \textbf{other religious minorities} (Islamophobia, antisemitism) — same framework, different case
          \item Apply to \textbf{more languages} (Korean, Vietnamese) to test whether moral-architecture finding generalises
          \item Use \textbf{larger LLMs} with RAG for more stable low-resource augmentation
          \item Longitudinal analysis: does moral framing \textbf{shift} after major events (e.g.\ 2022 Abe assassination)?
        \end{itemize}
      \end{block}
  \end{columns}
\end{frame}

% ── SLIDE 19 · Summary ───────────────────────────────────────
\begin{frame}{Summary: The Big Picture}

  \begin{center}
    \Large
    We started with a simple question:\\[8pt]
    \textit{``Why do people hate Christianity — and is it the same reason\\
    in English, Chinese, and Japanese?''}
  \end{center}

  \vspace{10pt}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{block}{The answer}
        \small
        \textbf{No} — the targets, the rhetoric, and the moral logic are structurally different.\\[4pt]
        English speakers condemn. East Asian speakers fear.\\[4pt]
        But the \emph{choice of weapon} (attack frame) follows the \emph{type of victim} — consistently, across all three languages.
      \end{block}
    \column{0.48\textwidth}
      \begin{alertblock}{The broader lesson}
        \small
        Binary hate speech detection cannot surface these differences.\\[4pt]
        \textbf{Structured, interpretable, cross-lingual analysis can.}\\[4pt]
        The who–how–why pipeline is a step toward that goal.
      \end{alertblock}
